实现多表分页查询

1.在dao类新建一个方法比如说 
List<Roominf>listByPage(IPage<Roominf> page,@Param("params") Map params);
其中声明返回Roominf的列表 添加IPage会自动执行分页查询操作 声明Map
@Param注解是MyBatis中用来传递参数的注解。在DAO接口的方法中，如果需要传递多个参数，
可以使用@Param注解为每个参数指定一个名称，以便在SQL语句中引用这些参数。
所以调用这个方法的时候需要传入page和map两个参数

2.在对应service层添加方法
Page<Roominf> listByPage(@Param("params") Map params);
page方法根据map也就是查询条件进行分页查询操作

3.在对应xml文件编写对应的sql语句
namespace配置DAO的路径，id为方法名,resultMap为这个模块取一个别名

<mapper namespace="com.example.house.generator.mapper.RoominfMapper">
<select id="listByPage" resultMap="BaseResultMap">

SELECT r.*,t.*,u.*,s.* FROM  roominf r  
JOIN type t ON r.zhonglei = t.id 
JOIN useto u ON r.yongtu = u.usetoid 
JOIN state s ON r.zhuangtai = s.stateid 
WHERE 1 = 1

//相等查询(if判断传入map对应值是否存在,其中params是dao方法定义的名字)
<if test="params.num != null and params.num !=''">
  and r.num = #{params.num}
</if>
//模糊查询
<if test="params.mingzi != null and params.mingzi !=''">
  and r.mingzi like concat('%',#{params.mingzi},'%')
</if>
......
</select>
定义了一个名为"BaseResultMap"的结果映射，用于将数据库查询结果映射到Java对象
com.example.house.generator.domain.Roominf 中。
<resultMap id="BaseResultMap" type="com.example.house.generator.domain.Roominf">
定义了一个名为 "BaseResultMap" 的结果映射，将查询结果映射到 Roominf 类型的对象。
<id> 和 <result> 标签：分别用于配置主键和普通字段的映射关系，指定了属性名、数据库列名
以及 JDBC 数据类型。
<association> 标签：用于配置关联对象的映射关系，这里包括了 Type、Useto 和 State 
三个关联对象的映射配置，指定了属性名、关联的 Java 类型以及各自的字段映射关系。
注：如此写法Roominf的bean必须添加private  Useto useto;对应bean

 <resultMap id="BaseResultMap" type="com.example.house.generator.domain.Roominf">
    <id property="num" column="num" jdbcType="INTEGER"/>
    <result property="weizhi" column="weizhi" jdbcType="VARCHAR"/>
    <association property="type" javaType="com.example.house.generator.domain.Type">
        <result property="id" column="id" />
        <result property="type" column="type" jdbcType="VARCHAR"/>
    </association>
    <association property="useto" javaType="com.example.house.generator.domain.Useto">

    </association>
    <association property="state" javaType="com.example.house.generator.domain.State">
        <result property="stateid" column="stateid"/>
        <result property="statename" column="statename" jdbcType="VARCHAR"/>
    </association>
</resultMap>
</mapper>

4.在impl编写调用方法
@Override
public Page<Roominf> listByPage(Map params) {
    //这两个参数为前端封装传入的数据
    current为当前页,size为一页显示多少条数据
    int current = (int) params.get("Current");
    int size = (int) params.get("size");
    //进行分页操作
    Page<Roominf> page = new Page<Roominf>(current,size);
    //根据查询条件进行查询
    List<Roominf> UsersList= roominfMapper.listByPage(page,params);
    page.setRecords(UsersList);
    return page;
}

 // 创建axios实例

 baseURL: process.env.VUE_APP_BASE_API , // api的base_url

 'Content-Type':'application/json'

// request拦截器
service.interceptors.request.use(
  config => {
    config.headers['token'] = Cookie.get("token")
    console.log("url:",config.url)
    return config
  },
  error => {
    console.log(error)
    return Promise.reject(error)
  }
)

// response拦截器
  service.interceptors.response.use(
    response => {
      const res = response.data
       const config = response.config
       // 二进制流处理
      if (config && config.responseType === 'blob') {
         return response
       }
       if (res.code === 401) {
         alert('登录状态已过期，请您重新登录');
         // this.$router.push('/login');
         router.replace('/login');
       }
       else if (res.code !== 200) {
        console.log('请求异常：')
        // this.$message('请求异常')
       } else {
        console.log('请求地址：', response.config.url)
      }
      return res
     },
    error => {
       console.log('请求异常：' + error)

    })

uniapp

uniapp配置

 1.打开HBuilder新建项目选择vue2
 2.封装请求方式添加utils包 新建wx-request.js
// 全局请求封装
  const base_url = 'http://localhost:8919/'
  // 需要修改token，和根据实际修改请求头
  export default (params) => {
    let url = params.url;
    let method = params.method || "get";
    let data = params.data || {};
    let header = {}
    if (method == "post") {
      header = {
        'Content-Type': 'application/json'
      };
    }
    // 获取本地token
    if (uni.getStorageSync("token")) {
      header['Authorization'] = 'Bearer ' + uni.getStorageSync("token");
    }

    return new Promise((resolve, reject) => {
      uni.request({
        url: base_url + url,
        method: method,
        header: header,
        data: data,
        success(response) {
          const res = response
          // 根据返回的状态码做出对应的操作
          //获取成功
          console.log(res.statusCode);
          if (res.statusCode == 200) {
            resolve(res.data);
          } else {
            uni.clearStorageSync()
            switch (res.statusCode) {
              case 401:
                uni.showModal({
                  title: "提示",
                  content: "请登录",
                  showCancel: false,
                  success(res) {
                    setTimeout(() => {
                      uni.navigateTo({
                        url: "/pages/login/index",
                      })
                    }, 1000);
                  },
                });
                break;
              case 404:
                uni.showToast({
                  title: '请求地址不存在...',
                  duration: 2000,
                })
                break;
              default:
                uni.showToast({
                  title: '请重试...',
                  duration: 2000,
                })
                break;
            }
          }
        },
        fail(err) {
          console.log(err)
          if (err.errMsg.indexOf('request:fail') !== -1) {
            wx.showToast({
              title: '网络异常',
              icon: "error",
              duration: 2000
            })
          } else {
            wx.showToast({
              title: '未知异常',
              duration: 2000
            })
          }
          reject(err);

        },
        complete() {
          // 不管成功还是失败都会执行
          uni.hideLoading();
          uni.hideToast();
        }
      });
    }).catch((e) => {});
  };
